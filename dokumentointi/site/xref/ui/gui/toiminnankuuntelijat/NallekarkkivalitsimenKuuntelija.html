1   package ui.gui.toiminnankuuntelijat;
2   
3   import java.awt.event.*;
4   import javax.swing.*;
5   import logiikka.Pelivelho;
6   import ui.gui.piirtaminen.Piirtoalusta;
7   
8   /**
9    * Nimensä mukaisesti.
10   *
11   * @author xvixvi
12   */
13  public class NallekarkkivalitsimenKuuntelija extends VuoronPaattavaKuuntelija implements ActionListener {
14  
15      private final Pelivelho pelivelho;
16      private final JLabel[] nallekarkkikentat;
17      private final JLabel infokentta;
18      private int[] nostettavat;
19  
20      /**
21       * Luo kuuntelijan.
22       * 
23       * @param pelivelho Veikkaapa.
24       * @param infokentta Veikkaapa.
25       * @param nallekarkkikentat Veikkaapa.
26       * @param pa piirtoavustaja.
27       * @param napit joita näytetään ja piilotetaan.
28       */
29      public NallekarkkivalitsimenKuuntelija(Pelivelho pelivelho, JLabel infokentta, JLabel[] nallekarkkikentat, Piirtoalusta pa, JComponent[] napit) {
30          super(pelivelho, napit, pa, infokentta);
31          this.pelivelho = pelivelho;
32          this.nallekarkkikentat = nallekarkkikentat;
33          this.infokentta = infokentta;
34      }
35  
36      @Override
37      public void actionPerformed(ActionEvent e) {
38          boolean kaksiSamanvarista = false;
39          int summa = 0;
40          int vari = 0;
41          nostettavat = new int[5];
42  
43          for (int i = 0; i < nallekarkkikentat.length; i++) {
44              JLabel nallekarkkikentta = nallekarkkikentat[i];
45  
46              int arvo = Integer.parseInt(nallekarkkikentta.getText());
47              nostettavat[i] = arvo;
48  
49              if (arvo == 2) {
50                  kaksiSamanvarista = true;
51                  vari = i + 1;
52              }
53  
54              summa += arvo;
55          }
56          tyhjennaKentat();
57  
58          if (tarkistasyote(kaksiSamanvarista, summa, vari)) {
59              return;
60          }
61  
62          pelivelho.nostaNallekarkkeja(nostettavat);
63          super.lopetaVuoro();
64      }
65  
66      private boolean tarkistasyote(boolean kaksiSamanvarista, int summa, int vari) {
67          boolean lopeta = false;
68          if (kaksiSamanvarista && summa > 2) {
69              infokentta.setText("Jos haluat kaksi samanväristä, et voi valita muita karkkeja.");
70              lopeta = true;
71          } else if (summa > 3) {
72              infokentta.setText("Et voi valita noin montaa nallekarkkia, senkin karvakorvainen humanisti!");
73              lopeta = true;
74          } else if (kaksiSamanvarista && !pelivelho.vahintaanNeljaKarkkia(vari)) {
75              infokentta.setText("Kasassa on oltava vähintään neljä karkkia, että voit nostaa kaksi samasta kasasta.");
76              lopeta = true;
77          }
78          return lopeta;
79      }
80  
81      private void tyhjennaKentat() {
82          for (JLabel kentta : nallekarkkikentat) {
83              kentta.setText("0");
84          }
85      }
86  
87  }
