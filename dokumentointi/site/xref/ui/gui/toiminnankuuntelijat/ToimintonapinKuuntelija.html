1   package ui.gui.toiminnankuuntelijat;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.*;
6   import logiikka.Pelivelho;
7   
8   /**
9    * Kuuntelee nappeja, joilla pelaaja päättää mitä tekee vuorollaan.
10   *
11   * @author xvixvi
12   */
13  public class ToimintonapinKuuntelija implements ActionListener {
14  
15      private final Pelivelho pelivelho;
16      private final JComponent[] napit;
17      private final String teksti;
18  
19      /**
20       * Luo kuuntelijan.
21       * 
22       * @param pelivelho Veikkaapa.
23       * @param n tämän napin teskti.
24       * @param napit napit, joiden näkyvyyttä säädetään.
25       */
26      public ToimintonapinKuuntelija(Pelivelho pelivelho, JButton n, JComponent[] napit) {
27          this.pelivelho = pelivelho;
28          this.napit = napit;
29          teksti = n.getText();
30      }
31  
32      @Override
33      public void actionPerformed(ActionEvent e) {
34          if (teksti.contains("Nostan")) {
35              napit[1].setVisible(true);
36              napit[3].setVisible(true);
37              napit[0].setVisible(false);
38          } else if (teksti.contains("takaisin")) {
39              napit[0].setVisible(true);
40              for (int i = 1; i < 4; i++) {
41                  napit[i].setVisible(false);
42              }
43          } else if (teksti.contains("Ostan")) {
44              napit[2].setVisible(true);
45              napit[3].setVisible(true);
46              napit[0].setVisible(false);
47              pelivelho.vieVarauksetJaNakyvatOmistuksetValitsimelle();
48          } else if (teksti.contains("Varaan")) {
49              napit[2].setVisible(true);
50              napit[3].setVisible(true);
51              napit[0].setVisible(false);
52              pelivelho.vieNakyvatOmistuksetValitsimelle();
53          }
54      }
55  }
