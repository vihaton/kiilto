73      public int[] mitaKarkkejaNostetaan(Kasakokoelma markkinat) {
74          int[] nosto = new int[5];
75          
76          while (true) {
77              System.out.println("Mitä nallekarkkeja haluaisit? ('o' ohjeet, 'e' takaisin)");
78              String syote = lukija.nextLine();
79              if (syote.equalsIgnoreCase("o")) {
80                  tulostaKarkkienNostamisOhjeet();
81              } else if (syote.equalsIgnoreCase("e")) {
82                  return null;
83              } else {
84                  try {
85                      nosto = tarkistaKarrkienNostamisSyote(syote, markkinat);
86                      break;
87                  } catch (Exception e) {
88                      System.out.println("Syötteesi oli huono.\n" + e.getMessage()+"\n");
89                  }
90              }
91          }
92          return nosto;
93      }

95      private int[] tarkistaKarrkienNostamisSyote(String syote, Kasakokoelma markkinat) throws Exception {
96          int[] nosto = new int[5];
97          int summa = 0;
98          boolean sama = false;
99          
100         String[] maarat = syote.split(",", 5);
101                     
102         for (int i = 0; i < 5; i++) {
103             int maara = Integer.parseInt(maarat[i].trim());
104             if (maara<0 || maara>2) throw new IllegalArgumentException("luvut voivat olla 0,1 tai 2.");
105             if (maara == 2) sama = true;
106             summa += maara;
107             if (summa > 3) throw new IllegalArgumentException("eipäs rohmuta!");
108             if (sama && summa > 2) throw new IllegalArgumentException("eipäs rohmuta!");
109             if (markkinat.getKasanKoko(i+1)< maara) throw new IllegalArgumentException("liian vähän karkkeja markkinoilla!");
110             if (maara == 2 && markkinat.getKasanKoko(i+1) < 4) throw new IllegalArgumentException("kasassa on liian vähän karkkeja kahden ottamiseen!");
111             //syöte on ollu tähän numeroon mennessä hyväksyttävä
112             nosto[i] = maara;
113         }
114         if (sama && summa!=2) throw new IllegalArgumentException("Jos otat kaksi samanväristä, saat vain yhtä väriä.");
115         if (!sama && summa !=3) throw new IllegalArgumentException("Et nostanut kolmea karkkia.");
116         
117         return nosto;
118     }

134     public int mikaOmistusOstetaan(ArrayList<String> nakyvatOmistukset) {
135         int omistuksenNro = 0;
136         System.out.println("Haluatko ostaa omistuksen pöydältä vai varauksistasi?\n"
137                 + "'v' -> varauksistasi, 'e' -> exit, kaikki muut syötteet -> pöydältä.");
138         String syote = lukija.nextLine();
139         if (syote.equalsIgnoreCase("v")) return 0;
140         if (syote.equalsIgnoreCase("e")) return -1;
141         
142         while (omistuksenNro==0) {
143             System.out.println("Anna ostettavan omistuksen numero:");
144             try {
145                 syote = lukija.nextLine();
146                 if (!nakyvatOmistukset.contains(syote.trim())) throw new IllegalArgumentException();
147                 omistuksenNro = Integer.parseInt(syote);
148             } catch (Exception e) {
149                 omistuksenNro = 0;
150                 System.out.println("Huono syöte, yritäppä uudestaan tahmanäppi!");
151             }
152         }
153         return omistuksenNro;
154     }
