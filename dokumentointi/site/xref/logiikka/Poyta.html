56      private void luoOmistukset(Scanner lukija) {
57          for (int i = 0; i < 3; i++) {
58              Omaisuus omistuspakka = omistuspakat.get(i);
59              int j = 40;
60              if (i == 1) j = 30;
61              if (i == 2) j = 20;
62              
63              while (j>0) {
64                  omistuspakka.lisaaOmistus(luoOmistus(lukija.nextLine()));
65                  j--;
66              }
67          }
68      }

90      private void luoMerkkihenkilot(Scanner lukija) {
91          for (int i = 0; i < 10; i++) {
92              Merkkihenkilo m = luoMerkkihenkilo(lukija.nextLine());
93              merkkihenkilot.add(m);
94          }
95      }

97      private Merkkihenkilo luoMerkkihenkilo(String rivi) {
98          String[] palat = rivi.split(",",9);
99          int[] vaatimus = new int[6];
100         for (int i = 2; i < 7; i++) {
101             vaatimus[i-2] = Integer.parseInt(palat[i]);
102         }
103         return new Merkkihenkilo(palat[0], vaatimus, Integer.parseInt(palat[1]));
104     }

106     private void valitsePelinMerkkihenkilot() {
107         int henkiloita = 5;
108         if (pelaajat.size() == 3) henkiloita = 4;
109         if (pelaajat.size() == 2) henkiloita = 2;
110         
111         while (merkkihenkilot.size() > henkiloita) {
112             int poistettava = arpoja.nextInt(merkkihenkilot.size());
113             merkkihenkilot.remove(poistettava);
114         }
115     }

118     @Override
119     public String toString() {
120         String s = "Pöydässä pelaajat:\n";
121         for (Pelaaja p : pelaajat) {
122             s = s.concat(p+"\n");
123         }
124         
125         s = s.concat("Merkkihenkilöt:\n");
126         for (int i = 0; i < merkkihenkilot.size(); i++) {
127             s = s.concat(merkkihenkilot.get(i).toString() + "\n");
128         }
129         
130         for (int i = omistuspakat.size()-1; i > -1; i--) {
131             s = s.concat("Omistuspakasta " + (i+1) +":\n\n");
132             Omaisuus oma = omistuspakat.get(i);
133             s = s.concat(oma.paallimmaisetToString()+"\n");
134         }
135         s = s.concat("***\n");
136         s = s.concat("karkkimarkkinat:\n" +karkkimarkkinat.toString() + "\n");
137         s = s.concat("***\n");
138         return s;
139     }

153     /**
154      * 
155      * @param pelaaja Joka ostaa omaisuutta pöydältä.
156      * @param ostonNumero Joka ostetaan.
157      * @return boolean Onnistuiko osto.
158      */
159     boolean suoritaOsto(Pelaaja pelaaja, int ostonNumero) {
160         Omaisuus omistuspakka;
161         if (!this.getNakyvienNimet().contains(""+ostonNumero)) return false;
162 
163         if (ostonNumero < 41) omistuspakka=omistuspakat.get(0);
164         else if (ostonNumero < 71) omistuspakka=omistuspakat.get(1);
165         else omistuspakka=omistuspakat.get(2);
166         
167         //kysytään pelaajalta, onko hänellä varaa ostaa nimeämänsä omistus
168         Omistus o = omistuspakka.getNakyvaOmistus(ostonNumero);
169         if (pelaaja.onkoVaraa(o)) {
170             pelaaja.osta(omistuspakka, o, karkkimarkkinat);
171             return true;
172         }
173         return false;
174     }

176     /**
177      * 
178      * @param pelaaja joka yrittää lunastaa varauksensa.
179      * @param varauksenNumero varaus, jota lunastetaan.
180      * @return Boolean onnistuiko yritys.
181      */
182     public boolean suoritaOstoVarauksista(Pelaaja pelaaja, int varauksenNumero) {
183         if (!pelaaja.getVaraustenNumerot().contains(""+varauksenNumero)) return false;
184         
185         Omistus o = pelaaja.getVaraus(varauksenNumero);
186         if (pelaaja.onkoVaraa(o)) {
187             return pelaaja.ostaVaraus(o, karkkimarkkinat);
188         }
189         return false;
190     }

192     /**
193      * 
194      * @param pelaaja
195      * @param varauksenNro varattavan varauksen numero (eli nimi).
196      * @return boolean, onnistuiko varaaminen.
197      */
198     boolean teeVaraus(Pelaaja pelaaja, int varauksenNro) {
199         if (pelaaja == null) return false;
200         if (pelaaja.getVarauksienMaara()>=3) return false;
201         int pakka = 0;
202         if (varauksenNro>40) pakka = 1;
203         if (varauksenNro>70) pakka = 2;
204         Omaisuus omistusPakka = omistuspakat.get(pakka);
205         
206         Omistus varattava = omistusPakka.getNakyvaOmistus(varauksenNro);
207         if (varattava == null) return false;
208         if (!pelaaja.teeVaraus(varattava)) return false;
209         omistusPakka.poistaOmistus(varattava);
210         annaPelaajalleKulta(pelaaja);
211 
212         return true;
213     }
