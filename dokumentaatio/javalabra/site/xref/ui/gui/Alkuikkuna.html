8   /**

14  public class Alkuikkuna implements Runnable {
15  
16      private JFrame alkuvalikko;

18  
19      /**
20       * Luo alkuikkunan, joka luo itselleen pelin pelivelhon.

24      }

26      @Override
27      public void run() {
28          alkuvalikko = new JFrame();
29          alkuvalikko.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
30          alkuvalikko.setPreferredSize(new Dimension(350, 400));
31          alkuvalikko.setLocation(500, 200);
32  
33          luoSisalto(alkuvalikko);

35          alkuvalikko.pack();

44      private void luoSisalto(Container container) {
45          GridLayout layout = new GridLayout(5, 0);
46          container.setLayout(layout);
47  
48          JLabel pelaajia = new JLabel("Kuinka monen pelaajan pelin haluaisit pelata?");
49          pelaajia.setHorizontalAlignment(JLabel.CENTER);
50  
51          final JButton kaksiPelaa = new JButton("Kaksin aina kaunihimpi");
52          final JButton kolmePelaa = new JButton("Kolmin selvästi kovempi");
53          final JButton neljaPelaa = new JButton("Nelistään, saa eniten nautintoa, pelistään");
54          final JButton lopeta = new JButton("Lopeta sekoilu, kiitos!");
55  
56          /*
57           WIP
58          
59           -mahdollisuus syöttää pelaajien nimet tekstikenttiin
60           */
61          lisaaKuuntelijat(kaksiPelaa, kolmePelaa, neljaPelaa, lopeta);
62  
63          container.add(pelaajia);
64          container.add(kaksiPelaa);
65          container.add(kolmePelaa);
66          container.add(neljaPelaa);
67          container.add(lopeta);
68      }
69  
70      private void lisaaKuuntelijat(JButton kaksiPelaa, JButton kolmePelaa, JButton neljaPelaa, JButton lopeta) {
71          kaksiPelaa.addActionListener(new ActionListener() {
72  
73              @Override
74              public void actionPerformed(ActionEvent ae) {
75                  alkuvalikko.setVisible(false);
76                  alkuvalikko.dispose();
77  
78                  pelivelho.luoPelaajat(2);
79                  pelivelho.pelaa();
80              }
81          });
82  
83          kolmePelaa.addActionListener(new ActionListener() {
84  
85              @Override
86              public void actionPerformed(ActionEvent ae) {
87                  alkuvalikko.setVisible(false);
88                  alkuvalikko.dispose();
89  
90                  pelivelho.luoPelaajat(3);
91                  pelivelho.pelaa();
92              }
93          });
94  
95          neljaPelaa.addActionListener(new ActionListener() {
96  
97              @Override
98              public void actionPerformed(ActionEvent ae) {
99                  alkuvalikko.setVisible(false);
100                 alkuvalikko.dispose();
101 
102                 pelivelho.luoPelaajat(4);
103                 pelivelho.pelaa();
104             }
105         });
106 
107         lopeta.addActionListener(new ActionListener() {
108             @Override
109             public void actionPerformed(ActionEvent ae) {
110                 System.exit(0);
111             }
112         });
113     }
